<script>
  const audio = document.getElementById('audio');
  const btn = document.getElementById('playBtn');
  const eq = document.getElementById('eq');
  const btnText = document.getElementById('btnText');

  function toggleMusic() {
    if (audio.paused) {
      audio.play().then(() => {
        btn.classList.add('playing');
        eq.classList.remove('paused');
        btnText.innerText = "إيقاف الأغنية ⏸️";
      }).catch(e => {
        console.error("Audio error:", e);
        alert("حدث خطأ أثناء تشغيل الصوت، تأكد من اتصال الإنترنت.");
      });
    } else {
      audio.pause();
      btn.classList.remove('playing');
      eq.classList.add('paused');
      btnText.innerText = "تشغيل الأغنية ❤️";
    }
  }

  function explode(e) {
    const items = ['❤️','🌸','🌹','🎂','✨'];
    for(let i=0; i<50; i++) {
      const el = document.createElement('div');
      el.className = 'fly-item';
      el.innerHTML = items[Math.floor(Math.random()*items.length)];
      el.style.left = e.clientX + 'px';
      el.style.top = e.clientY + 'px';
      el.style.setProperty('--dx', (Math.random()-0.5)*400 + 'px');
      el.style.setProperty('--rot', Math.random()*720 + 'deg');
      document.body.appendChild(el);
      setTimeout(() => el.remove(), 1500);
    }
  }

  const bg = document.getElementById('bgLayer');
  for(let i=0; i<40; i++) {
    const h = document.createElement('div');
    h.className = 'bg-heart';
    h.innerHTML = ['💖','🌸','✨'][Math.floor(Math.random()*3)];
    h.style.left = Math.random()*100 + 'vw';
    h.style.animationDuration = 5 + Math.random()*10 + 's';
    h.style.animationDelay = -Math.random()*10 + 's';
    bg.appendChild(h);
  }
</script>
