<script>
  /* ===== BG HEARTS GENERATOR ===== */
  const bgSymbols = ['💖','🌸','💗','🌹','💕','✨','💓','🌷','💝','⭐'];
  const bgLayer = document.getElementById('bgLayer');

  for (let i = 0; i < 50; i++) {
    const el = document.createElement('div');
    el.className = 'bg-heart';
    el.innerHTML = bgSymbols[Math.floor(Math.random() * bgSymbols.length)];
    el.style.left = Math.random() * 100 + 'vw';
    el.style.fontSize = (14 + Math.random() * 20) + 'px';
    el.style.animationDuration = (7 + Math.random() * 12) + 's';
    el.style.animationDelay = -(Math.random() * 15) + 's';
    bgLayer.appendChild(el);
  }

  /* ===== SPARKLES ===== */
  for (let i = 0; i < 20; i++) {
    const sp = document.createElement('div');
    sp.className = 'sparkle';
    const size = 4 + Math.random() * 6;
    sp.style.cssText = `
      width:${size}px; height:${size}px;
      left:${Math.random()*100}vw; top:${Math.random()*100}vh;
      animation-duration:${2 + Math.random()*4}s;
      animation-delay:${Math.random()*4}s;
    `;
    document.body.appendChild(sp);
  }

  /* ===== LOCAL AUDIO PLAYER SETUP ===== */
  const audio = document.getElementById('myAudio');
  const playBtn = document.getElementById('playBtn');
  const btnText = document.getElementById('btnText');
  const eq = document.getElementById('eq');

  function toggleSong() {
    if (audio.paused) {
      audio.play().then(() => {
        playBtn.classList.add('playing');
        btnText.textContent = 'إيقاف الأغنية ⏸️';
        eq.classList.remove('paused');
      }).catch((error) => {
        console.log("Audio play failed:", error);
        btnText.textContent = 'اضغط مرة أخرى ⚠️';
      });
    } else {
      audio.pause();
      playBtn.classList.remove('playing');
      btnText.textContent = 'تشغيل الأغنية ❤️';
      eq.classList.add('paused');
    }
  }

  /* ===== EXPLODE LOVE (HEARTS, FLOWERS & CAKE) ===== */
  const flyItems = ['❤️', '💜', '💙', '💛', '🌸', '🌹', '🎂'];

  function explodeLove(e) {
    const cx = e.clientX || window.innerWidth / 2;
    const cy = e.clientY || window.innerHeight / 2;

    for (let i = 0; i < 60; i++) {
      setTimeout(() => createFlyItem(cx, cy), i * 15);
    }
    
    if (navigator.vibrate) {
      navigator.vibrate(50);
    }
  }

  function createFlyItem(cx, cy) {
    const el = document.createElement('div');
    el.className = 'fly-item';
    el.innerHTML = flyItems[Math.floor(Math.random() * flyItems.length)];

    const dx = (Math.random() - 0.5) * 400; 
    const rot = (Math.random() - 0.5) * 720; 
    const dur = 1.2 + Math.random() * 1.5;   
    const size = 20 + Math.random() * 30;    

    el.style.left = cx + 'px';
    el.style.top = cy + 'px';
    el.style.fontSize = size + 'px';
    el.style.setProperty('--dx', dx + 'px');
    el.style.setProperty('--rot', rot + 'deg');
    el.style.animationDuration = dur + 's';

    document.body.appendChild(el);
    
    setTimeout(() => el.remove(), dur * 1000 + 100);
  }

  /* ===== CURSOR SPARKLE ===== */
  document.addEventListener('touchmove', createCursorSparkle);
  document.addEventListener('mousemove', createCursorSparkle);

  function createCursorSparkle(e) {
    if (Math.random() > 0.85) {
      const s = document.createElement('div');
      const clientX = e.touches ? e.touches[0].clientX : e.clientX;
      const clientY = e.touches ? e.touches[0].clientY : e.clientY;
      
      s.style.cssText = `
        position:fixed; pointer-events:none; z-index:9999;
        left:${clientX}px; top:${clientY}px;
        font-size:${12 + Math.random() * 14}px;
        animation: flyUp 1s ease forwards;
        --dx:${(Math.random() - 0.5) * 60}px;
        --rot:${Math.random() * 360}deg;
      `;
      s.className = 'fly-item';
      s.innerHTML = Math.random() > 0.5 ? '✨' : '🌸';
      document.body.appendChild(s);
      setTimeout(() => s.remove(), 1000);
    }
  }
</script>
